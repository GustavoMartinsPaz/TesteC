#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

// don't touch
typedef char* string;

void print_bulb(int bit) {
    if (bit == 0) {
      // Dark emoji
      printf("\U000026AB");
    }
    else if (bit == 1) {
      // Light emoji
      printf("\U0001F7E1");
    }
}


int main(int argc, string argv[]) {

  if (argc != 2) {
    printf("Usage: %s <filename>\n", argv[0]);
    return 1;
  }
    
  char *filename = argv[1];
  FILE *fp = fopen(filename, "r");
  
  if (!fp) {
    printf("Error: cannot open file %s\n", filename);
    return 1;
  }
  
  int n;
  fscanf(fp, "%d\n", &n);
     
  char line[256];
  for (int i = 0; i < n; i++) {
    if (fgets(line, sizeof(line), fp)) {

      line[strcspn(line, "\n")] = '\0';
        for (int j = 0; line[j] != '\0'; j++) {
          int c = line[j];

            for(int k = 7; k >=0; k--) {
              int bit = (c >> k) & 1;
              print_bulb(bit);
            }

          printf("\n");
        }
        
      printf("\n");
    }
  }
    
  fclose(fp);
  return 0;
}
